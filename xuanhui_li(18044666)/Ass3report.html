# --- 超参数调优 ---
# 1. 定义要尝试的参数网格
#    这里选择三个关键参数进行调优：
#    n_estimators: 树的数量
#    learning_rate: 学习率
#    max_depth: 每棵树的最大深度
param_grid = {
    'n_estimators': [100, 200, 300],
    'learning_rate': [0.05, 0.1, 0.2],
    'max_depth': [3, 5, 7]
}
# --- 创建一个基础模型，GridSearchCV会用它来测试不同的参数 ---
base_model = xgb.XGBRegressor(objective='reg:squarederror', random_state=42)
# --- 设置时间序列交叉验证 ---
# --- n_splits=5 （创建5个连续的训练/验证数据折叠） ---
tscv = TimeSeriesSplit(n_splits=5)
# --- 初始化并运行GridSearchCV ---
#    scoring='neg_root_mean_squared_error' 表示目标是最小化RMSE
#    n_jobs=-1 会使用所有可用的CPU核心来加速计算
#    verbose=2 会打印出搜索过程
grid_search = GridSearchCV(
    estimator=base_model,
    param_grid=param_grid,
    scoring='neg_root_mean_squared_error',
    cv=tscv,
    n_jobs=-1,
    verbose=2
)
# --- 在训练数据上运行搜索 ---
grid_search.fit(X_train, y_train)
print(f"找到的最佳参数组合是: {grid_search.best_params_}")
print(f"对应的最佳RMSE分数是: {-grid_search.best_score_:.4f}")
